常用的centos命令

echo

echo helloworld
echo $SHELL
    /bin/bash

date

#查看系统时间
[hj@hj-pc ~]$ date
2023年 11月 27日 星期一 21:57:02 CST
[hj@hj-pc ~]$ date "+%Y-%m-%d %H:%M:%S"
2023-11-27 21:58:57
#set
[root@hj-pc hj]# date -s "20200202 12:00:00"
2020年 02月 02日 星期日 12:00:00 CST
[root@hj-pc hj]# date "+%j"
033

timedatectl

参数：status    list-timezones      set-timezone    set-time
timedatectl status
timedatectl list-timezones
timedatectl set-timezone Asia/Shanghai
timedatectl set-time "20200505 12:12:12"

reboot

poweroff

wget

-b 后台下载
-P 下载到指定目录
-p 下载页面所有资源，包括图片链接
-r 递归下载
-c 断点续传
-t 最大尝试次数
wget -t 3 -r -p https://www.xxx.com

ps

查看系统进程状态
参数 -aux
R  S  D  Z  T
运行 中断 不可中断 僵尸 停止

调整进程的优先级 -20～19
